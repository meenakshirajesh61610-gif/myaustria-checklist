// State
let categories = [];
let activeCategory = 'all';
let activeFilter = 'all';
let stateMap = {}; // {itemId: {purchased, packed}}

function loadData() {
  try {
    const saved = localStorage.getItem('austria_checklist_v2');
    if (saved) {
      const parsed = JSON.parse(saved);
      categories = parsed.categories;
      stateMap = parsed.stateMap || {};
    } else {
      resetToDefault();
    }
  } catch(e) { resetToDefault(); }
}

function resetToDefault() {
  categories = JSON.parse(JSON.stringify(DEFAULT_DATA)).map(cat => ({
    ...cat,
    items: cat.items.map(i => ({ id: i.id, name: i.name }))
  }));
  stateMap = {};
}

function saveData() {
  localStorage.setItem('austria_checklist_v2', JSON.stringify({ categories, stateMap }));
}

function getState(id) {
  return stateMap[id] || { purchased: false, packed: false };
}

function toggle(itemId, field) {
  if (!stateMap[itemId]) stateMap[itemId] = { purchased: false, packed: false };
  stateMap[itemId][field] = !stateMap[itemId][field];
  saveData();
  render();
}

function deleteItem(catId, itemId) {
  const cat = categories.find(c => c.id === catId);
  if (!cat) return;
  cat.items = cat.items.filter(i => i.id !== itemId);
  delete stateMap[itemId];
  saveData();
  render();
}

function addItem() {
  const name = document.getElementById('newItemName').value.trim();
  const catId = document.getElementById('newItemCat').value;
  if (!name || !catId) return;
  const cat = categories.find(c => c.id === catId);
  if (!cat) return;
  const newId = 'custom_' + Date.now();
  cat.items.push({ id: newId, name });
  saveData();
  closeForm();
  render();
}

function setFilter(f, btn) {
  activeFilter = f;
  document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  render();
}

function setCat(catId) {
  activeCategory = catId;
  render();
}

function toggleForm() {
  const form = document.getElementById('addForm');
  const fab = document.getElementById('fabBtn');
  if (form.classList.contains('open')) {
    closeForm();
  } else {
    form.classList.add('open');
    fab.style.display = 'none';
    populateCatSelect();
    document.getElementById('newItemName').focus();
  }
}

function closeForm() {
  document.getElementById('addForm').classList.remove('open');
  document.getElementById('fabBtn').style.display = 'flex';
  document.getElementById('newItemName').value = '';
  document.getElementById('newItemCat').value = '';
  document.getElementById('addConfirm').disabled = true;
}

function checkAddBtn() {
  const name = document.getElementById('newItemName').value.trim();
  const cat = document.getElementById('newItemCat').value;
  document.getElementById('addConfirm').disabled = !(name && cat);
}

function populateCatSelect() {
  const sel = document.getElementById('newItemCat');
  sel.innerHTML = '<option value="">Select category...</option>';
  categories.forEach(cat => {
    const opt = document.createElement('option');
    opt.value = cat.id;
    opt.textContent = cat.name;
    sel.appendChild(opt);
  });
}

function buildCatNav() {
  const row = document.getElementById('catRow');
  row.innerHTML = '';
  const allBtn = document.createElement('button');
  allBtn.className = 'cat-btn' + (activeCategory === 'all' ? ' active' : '');
  allBtn.textContent = 'All';
  if (activeCategory === 'all') allBtn.style.background = '#fff';
  allBtn.onclick = () => setCat('all');
  row.appendChild(allBtn);

  categories.forEach(cat => {
    const btn = document.createElement('button');
    btn.className = 'cat-btn' + (activeCategory === cat.id ? ' active' : '');
    if (activeCategory === cat.id) btn.style.background = cat.color;
    btn.textContent = cat.name.split(' ').slice(0,2).join(' ');
    btn.onclick = () => setCat(cat.id);
    row.appendChild(btn);
  });
}

function render() {
  const search = document.getElementById('searchInput').value.toLowerCase();
  buildCatNav();

  // Compute totals
  const allItems = categories.flatMap(c => c.items);
  const total = allItems.length;
  const pCount = allItems.filter(i => getState(i.id).purchased).length;
  const pkCount = allItems.filter(i => getState(i.id).packed).length;
  const dCount = allItems.filter(i => getState(i.id).purchased && getState(i.id).packed).length;

  document.getElementById('bar-purchased').style.width = total ? (pCount/total*100)+'%' : '0%';
  document.getElementById('bar-packed').style.width = total ? (pkCount/total*100)+'%' : '0%';
  document.getElementById('bar-done').style.width = total ? (dCount/total*100)+'%' : '0%';
  document.getElementById('count-purchased').textContent = pCount+'/'+total;
  document.getElementById('count-packed').textContent = pkCount+'/'+total;
  document.getElementById('count-done').textContent = dCount+'/'+total;

  const container = document.getElementById('listContainer');
  // Keep emptyMsg
  const emptyMsg = document.getElementById('emptyMsg');
  container.innerHTML = '';
  container.appendChild(emptyMsg);

  let anyVisible = false;

  categories.forEach(cat => {
    if (activeCategory !== 'all' && activeCategory !== cat.id) return;

    const filteredItems = cat.items.filter(item => {
      const state = getState(item.id);
      const isDone = state.purchased && state.packed;
      const matchSearch = item.name.toLowerCase().includes(search);
      const matchFilter =
        activeFilter === 'all' ||
        (activeFilter === 'pending' && (!state.purchased || !state.packed)) ||
        (activeFilter === 'done' && isDone);
      return matchSearch && matchFilter;
    });

    if (!filteredItems.length) return;
    anyVisible = true;

    const catDone = filteredItems.filter(i => { const s=getState(i.id); return s.purchased&&s.packed; }).length;

    const block = document.createElement('div');
    block.className = 'category-block';

    // Header
    const hdr = document.createElement('div');
    hdr.className = 'cat-header';
    hdr.style.borderBottomColor = cat.color;
    hdr.innerHTML = `<span class="cat-title">${cat.name}</span>
      <span class="cat-badge" style="background:${cat.color}">${catDone}/${filteredItems.length}</span>`;
    block.appendChild(hdr);

    // Col headers
    const cols = document.createElement('div');
    cols.className = 'col-headers';
    cols.innerHTML = '<span>Item</span><span>Purchased</span><span>Packed</span><span></span>';
    block.appendChild(cols);

    filteredItems.forEach(item => {
      const state = getState(item.id);
      const isDone = state.purchased && state.packed;

      const row = document.createElement('div');
      row.className = 'item-row' + (isDone ? ' done' : '');

      // Name
      const nameEl = document.createElement('span');
      nameEl.className = 'item-name';
      nameEl.textContent = (isDone ? '✓ ' : '') + item.name;
      row.appendChild(nameEl);

      // Purchased btn
      const pCell = document.createElement('div');
      pCell.className = 'check-cell';
      const pBtn = document.createElement('button');
      pBtn.className = 'chk';
      pBtn.style.border = `2px solid ${state.purchased ? '#f4a261' : 'rgba(255,255,255,0.3)'}`;
      pBtn.style.background = state.purchased ? '#f4a261' : 'transparent';
      pBtn.textContent = state.purchased ? '✓' : '';
      pBtn.onclick = () => toggle(item.id, 'purchased');
      pCell.appendChild(pBtn);
      row.appendChild(pCell);

      // Packed btn
      const pkCell = document.createElement('div');
      pkCell.className = 'check-cell';
      const pkBtn = document.createElement('button');
      pkBtn.className = 'chk';
      pkBtn.style.border = `2px solid ${state.packed ? '#2a9d8f' : 'rgba(255,255,255,0.3)'}`;
      pkBtn.style.background = state.packed ? '#2a9d8f' : 'transparent';
      pkBtn.style.color = '#fff';
      pkBtn.textContent = state.packed ? '✓' : '';
      pkBtn.onclick = () => toggle(item.id, 'packed');
      pkCell.appendChild(pkBtn);
      row.appendChild(pkCell);

      // Delete
      const delBtn = document.createElement('button');
      delBtn.className = 'del-btn';
      delBtn.textContent = '×';
      delBtn.title = 'Remove item';
      delBtn.onclick = () => deleteItem(cat.id, item.id);
      row.appendChild(delBtn);

      block.appendChild(row);
    });

    container.appendChild(block);
  });

  emptyMsg.style.display = anyVisible ? 'none' : 'block';
}
